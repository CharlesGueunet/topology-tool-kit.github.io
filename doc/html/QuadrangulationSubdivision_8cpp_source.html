242  tmp[0] = static_cast<size_t>(q.i);

247  } else if(q.l == _a) {

259  Point pc = outputPoints_[t[1]];

263  // ab, ac vectors
264  Point ab = pb - pa;

266  // compute ab ^ ac
267  Geometry::crossProduct(&ab.x, &ac.x, &crossP.x);
268  // magnitude
269  auto mag = Geometry::magnitude(&crossP.x);
270  // ensure normal not null
271  if(mag > powf(10, -FLT_DIG)) {
272  // unitary normal vector
273  normals.emplace_back(crossP / mag);
274  }
275  }
276

278  for(size_t i = 1; i < normals.size(); ++i) {
279  auto dotProd = Geometry::dotProduct(&normals[0].x, &normals[i].x);
280  if(dotProd < 0.0F) {
281  normals[i] = normals[i] * -1.0F;
282  }
283  }
284 
285  if(!normals.empty()) {
286  // compute mean of normals
287  quadNormal = std::accumulate(
288  normals.begin(), normals.end(), Point{}, std::plus<Point>());
289  quadNormal = quadNormal / static_cast<float>(normals.size());
290  } else {
291  // set error value directly in output variable...
292  quadNormal.x = NAN;
293  }
294 
295  return quadNormal;
296 }

680 void ttk::QuadrangulationSubdivision::quadStatistics() {
681  Timer t;
682 
683  quadArea_.clear();
684  quadArea_.resize(outputQuads_.size());
685  quadDiagsRatio_.clear();
686  quadDiagsRatio_.resize(outputQuads_.size());
687  quadEdgesRatio_.clear();
688  quadEdgesRatio_.resize(outputQuads_.size());
689  quadAnglesRatio_.clear();
690  quadAnglesRatio_.resize(outputQuads_.size());
691  hausdorff_.clear();
692  hausdorff_.resize(outputPoints_.size());
693 
694 #ifdef TTK_ENABLE_OPENMP
695 #pragma omp parallel for num_threads(threadNumber_)
696 #endif // TTK_ENABLE_OPENMP
697  for(size_t i = 0; i < outputQuads_.size(); ++i) {
698  const auto &q = outputQuads_[i];
699  Point pi = outputPoints_[q.i];
700  Point pj = outputPoints_[q.j];
701  Point pk = outputPoints_[q.k];
702  Point pl = outputPoints_[q.l];
703 
704  // quadrangle area
705  float area0{}, area1{};
706  Geometry::computeTriangleArea(&pi.x, &pj.x, &pk.x, area0);
707  Geometry::computeTriangleArea(&pi.x, &pk.x, &pl.x, area1);
708  quadArea_[i] = area0 + area1;
709 
710  // diagonals ratio
711  auto diag0 = Geometry::distance(&pi.x, &pk.x);
712  auto diag1 = Geometry::distance(&pj.x, &pl.x);
713  quadDiagsRatio_[i] = std::min(diag0, diag1) / std::max(diag0, diag1);
714 
715  // edges ratio
716  std::array<float, 4> edges{
717  Geometry::distance(&pi.x, &pj.x), // ij
718  Geometry::distance(&pj.x, &pk.x), // jk
719  Geometry::distance(&pk.x, &pl.x), // kl
720  Geometry::distance(&pl.x, &pi.x), // li
721  };
722  quadEdgesRatio_[i] = *std::min_element(edges.begin(), edges.end())
723  / *std::max_element(edges.begin(), edges.end());
724 
725  // angles ratio
726  std::array<float, 4> angles{
727  Geometry::angle(&pi.x, &pl.x, &pi.x, &pj.x), // lij
728  Geometry::angle(&pj.x, &pi.x, &pj.x, &pk.x), // ijk
729  Geometry::angle(&pk.x, &pj.x, &pk.x, &pl.x), // jkl
730  Geometry::angle(&pl.x, &pk.x, &pl.x, &pi.x), // kli
731  };
732  quadAnglesRatio_[i] = *std::min_element(angles.begin(), angles.end())
733  / *std::max_element(angles.begin(), angles.end());
734  }
735 
736  // compute ratio between quad area and mean quad area
737 
738  // global surface area
739  float sumArea{};
740  for(const auto a : quadArea_) {
741  sumArea += a;
742  }
743  for(auto &a : quadArea_) {
744  a *= quadArea_.size() / sumArea;
745  }
746 
747  // compute the minimal distance from every triangulation point to
748  // every quadrangulation point
749 
750  std::vector<size_t> triVertsDist(vertexNumber_);
751 
752 #ifdef TTK_ENABLE_OPENMP
753 #pragma omp parallel for num_threads(threadNumber_)
754 #endif // TTK_ENABLE_OPENMP
755  for(size_t i = 0; i < static_cast<size_t>(vertexNumber_); ++i) {
756  float minDist{std::numeric_limits<float>::infinity()};
757  Point p{};
758  triangulation_->getVertexPoint(i, p.x, p.y, p.z);
759 
760  for(size_t j = 0; j < outputPoints_.size(); ++j) {
761  auto dist = Geometry::distance(&p.x, &outputPoints_[j].x);
762  if(dist < minDist) {
763  minDist = dist;
764  triVertsDist[i] = j;
765  }
766  }
767  }
768 
769  // compute triangulation bounding box diagonal
770  Point pmin{std::numeric_limits<float>::infinity(),
771  std::numeric_limits<float>::infinity(),
772  std::numeric_limits<float>::infinity()};
773  Point pmax{-std::numeric_limits<float>::infinity(),
774  -std::numeric_limits<float>::infinity(),
775  -std::numeric_limits<float>::infinity()};
776 
777  for(size_t i = 0; i < static_cast<size_t>(vertexNumber_); ++i) {
778  Point p{};
779  triangulation_->getVertexPoint(i, p.x, p.y, p.z);
780  pmax.x = std::max(pmax.x, p.x);
781  pmax.y = std::max(pmax.y, p.y);
782  pmax.z = std::max(pmax.z, p.z);
783  pmin.x = std::min(pmin.x, p.x);
784  pmin.y = std::min(pmin.y, p.y);
785  pmin.z = std::min(pmin.z, p.z);
786  }
787 
788  auto bboxDiag = Geometry::distance(&pmin.x, &pmax.x);
789 
790 #ifdef TTK_ENABLE_OPENMP
791 #pragma omp parallel for num_threads(threadNumber_)
792 #endif // TTK_ENABLE_OPENMP
793  for(size_t i = 0; i < outputPoints_.size(); ++i) {
794  float maxDist{};
795  for(size_t j = 0; j < static_cast<size_t>(vertexNumber_); ++j) {
796  Point p{};
797  triangulation_->getVertexPoint(j, p.x, p.y, p.z);
798 
799  if(triVertsDist[j] == i) {
800  auto dist = Geometry::distance(&p.x, &outputPoints_[i].x);
801  if(dist > maxDist) {
802  maxDist = dist;
803  }
804  }
805  }
806  hausdorff_[i] = maxDist / bboxDiag / vertexNumber_ * 1e8;
807  }
808 
809  std::stringstream msg;
810  msg << MODULE_S "Computed quad statistics in " << t.getElapsedTime() << " s."
811  << std::endl;
812  dMsg(std::cout, msg.str(), detailedInfoMsg);
813 }

827 // main routine
828 int ttk::QuadrangulationSubdivision::execute() {
829 
830  {
831  std::stringstream msg;
832  msg << MODULE_S "Beginning computation... " << std::endl;
833  dMsg(std::cout, msg.str(), infoMsg);
834  }
835 
836  Timer t;
837 
838  // clear output variables
839  clearData();
840 
841  // store input points (MSC critical points)
842  for(size_t i = 0; i < inputVertexNumber_; i++) {
843  outputPoints_.emplace_back(inputVertices_[i]);
844  }
845 
846  // copy input quads into vector
847  for(size_t i = 0; i < inputQuadNumber_; i++) {
848  outputQuads_.emplace_back(inputQuads_[i]);
849  }
850 
851  // fill outputInfos_ with input data (critical points)
852  outputVertType_.resize(outputPoints_.size());
853  std::fill(outputVertType_.begin(), outputVertType_.end(), 0);
854 
855  // fill outputSubdivision with input data
856  outputSubdivision_.resize(outputPoints_.size());
857  std::fill(outputSubdivision_.begin(), outputSubdivision_.end(), 0);
858 
859  // vertices to filter from relaxation, projection
860  std::set<size_t> filtered{};
861  if(!lockAllInputVertices) {
862  if(lockInputExtrema) {
863  // get extraordinary vertices
864  findExtraordinaryVertices(filtered);
865  }
866  } else {
867  // fill vector with all input points indices from 0 to inputVertexNumber_
868  for(size_t i = 0; i < inputVertexNumber_; ++i) {
869  filtered.insert(i);
870  }
871  }
872 
873  // main loop
874  for(size_t i = 0; i < subdivisionLevel_; i++) {
875  // subdivise each quadrangle by creating five new points, at the
876  // center of each edge (4) and at the barycenter of the four
877  // vertices (1).
878  subdivise();
879  }
880 
881  // retrieve mapping between every vertex and its neighbors
882  quadNeighbors_.clear();
883  quadNeighbors_.resize(outputPoints_.size());
884  getQuadNeighbors(outputQuads_, quadNeighbors_);
885 
886  // "relax" the new points, i.e. replace it by the barycenter of its
887  // four neighbors
888  for(size_t i = 0; i < relaxationIterations_; i++) {
889  relax(filtered);
890 
891  // project all points on the nearest triangle (except MSC critical
892  // points)
893  project(filtered, (i == relaxationIterations_ - 1));
894  }
895 
896  // compute valence of every quadrangle vertex
897  outputValences_.resize(outputPoints_.size());
898  std::transform(
899  quadNeighbors_.begin(), quadNeighbors_.end(), outputValences_.begin(),
900  [&](const std::set<size_t> &neighbors) { return neighbors.size(); });
901 
902  quadStatistics();
903 
904  bool criterion = false;
905  for(size_t i = 0; i < outputPoints_.size(); ++i) {
906  if(outputValences_[i] > 4) {
907  continue;
908  }
909  if(hausdorff_[i] > hausdorffLevel_) {
910  criterion = true;
911  break;
912  }
913  }
914 
915  if(criterion) {
916  // log, clean & early return
917  std::stringstream msg;
918  msg << MODULE_S "Error: the output quadrangulation exceeds the provided"
919  << " Haussdorff distance tolerance." << std::endl;
920  dMsg(std::cout, msg.str(), infoMsg);
921  if(!showResError_) {
922  clearData();
923  return 1;
924  }
925  }
926 
927  {
928  std::stringstream msg;
929  msg << MODULE_S "Produced " << outputQuads_.size() << " quadrangles with "
930  << outputPoints_.size() << " points in " << t.getElapsedTime()
931  << " s. (" << threadNumber_ << " thread(s))." << std::endl;
932  dMsg(std::cout, msg.str(), infoMsg);
933  }
934 
935  return 0;
936 }
