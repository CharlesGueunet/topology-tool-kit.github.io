504 int ttk::AbstractMorseSmaleComplex::setSeparatrices1(const
505 std::vector<Separatrix>& separatrices,
506  const std::vector<std::vector<Cell>>& separatricesGeometry) const{
507  const dataType* const scalars=static_cast<dataType*>(inputScalarField_);
508  std::vector<dataType>* outputSeparatrices1_cells_separatrixFunctionMaxima=
509  static_cast<std::vector<dataType>*>(outputSeparatrices1_cells_separatrixFunctionMaxima_);
510  std::vector<dataType>* outputSeparatrices1_cells_separatrixFunctionMinima=
511  static_cast<std::vector<dataType>*>(outputSeparatrices1_cells_separatrixFunctionMinima_);
512  std::vector<dataType>* outputSeparatrices1_cells_separatrixFunctionDiffs=
513  static_cast<std::vector<dataType>*>(outputSeparatrices1_cells_separatrixFunctionDiffs_);
514 
515  int pointId=(*outputSeparatrices1_numberOfPoints_);
516  int cellId=(*outputSeparatrices1_numberOfCells_);
517  int separatrixId=0;
518  if(outputSeparatrices1_cells_separatrixIds_->size()){
519  separatrixId=*std::max_element(outputSeparatrices1_cells_separatrixIds_->begin(),
520  outputSeparatrices1_cells_separatrixIds_->end())+1;
521  }
522 
523  const int dimensionality=inputTriangulation_->getCellVertexNumber(0)-1;
524 
525  for(const Separatrix& separatrix : separatrices){
526  if(!separatrix.isValid_) continue;
527  if(!separatrix.geometry_.size()) continue;
528 
529  const Cell& saddle=separatrix.source_;
530  const Cell& extremum=separatrix.destination_;
531 
532  // get separatrix type
533  const char separatrixType=std::min(extremum.dim_, dimensionality-1);
534 
535  // compute separatrix function diff
536  const dataType separatrixFunctionMaximum=std::max(discreteGradient_.scalarMax<dataType>(saddle, scalars),
537  discreteGradient_.scalarMax<dataType>(extremum, scalars));
538  const dataType separatrixFunctionMinimum=std::min(discreteGradient_.scalarMin<dataType>(saddle, scalars),
539  discreteGradient_.scalarMin<dataType>(extremum, scalars));
540  const dataType separatrixFunctionDiff=separatrixFunctionMaximum-separatrixFunctionMinimum;
541 
542  // get boundary condition
543  const char isOnBoundary=(char)discreteGradient_.isBoundary(saddle) + (char)discreteGradient_.isBoundary(extremum);
544 
545  bool isFirst=true;
546  for(const int geometryId : separatrix.geometry_){
547  int oldPointId=-1;
548  for(auto cellIte=separatricesGeometry[geometryId].begin(); cellIte!=separatricesGeometry[geometryId].end(); ++cellIte){
549  const Cell& cell=*cellIte;
550 
551  float point[3];
552  discreteGradient_.getCellIncenter(cell, point);
553 
554  outputSeparatrices1_points_->push_back(point[0]);
555  outputSeparatrices1_points_->push_back(point[1]);
556  outputSeparatrices1_points_->push_back(point[2]);
557 
558  if(cellIte==separatricesGeometry[geometryId].begin() or
559  cellIte==separatricesGeometry[geometryId].end()-1)
560  outputSeparatrices1_points_smoothingMask_->push_back(0);
561  else
562  outputSeparatrices1_points_smoothingMask_->push_back(1);
563  outputSeparatrices1_points_cellDimensions_->push_back(cell.dim_);
564  outputSeparatrices1_points_cellIds_->push_back(cell.id_);
565 
566  if(oldPointId!=-1){
567  outputSeparatrices1_cells_->push_back(2);
568  outputSeparatrices1_cells_->push_back(oldPointId);
569  outputSeparatrices1_cells_->push_back(pointId);
570 
571  outputSeparatrices1_cells_sourceIds_->push_back(saddle.id_);
572  outputSeparatrices1_cells_destinationIds_->push_back(extremum.id_);
573  outputSeparatrices1_cells_separatrixIds_->push_back(separatrixId);
574  outputSeparatrices1_cells_separatrixTypes_->push_back(separatrixType);
575  outputSeparatrices1_cells_separatrixFunctionMaxima->push_back(separatrixFunctionMaximum);
576  outputSeparatrices1_cells_separatrixFunctionMinima->push_back(separatrixFunctionMinimum);
577  outputSeparatrices1_cells_separatrixFunctionDiffs->push_back(separatrixFunctionDiff);
578  outputSeparatrices1_cells_isOnBoundary_->push_back(isOnBoundary);
579 
580  ++cellId;
581  isFirst=false;
582  }
583 
584  oldPointId=pointId;

586  }
587  }
588 
589  if(!isFirst)
590  ++separatrixId;
591  }
592 
593  (*outputSeparatrices1_numberOfPoints_)=pointId;
594  (*outputSeparatrices1_numberOfCells_)=cellId;
595 
596  return 0;
