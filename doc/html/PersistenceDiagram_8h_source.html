127 int PersistenceDiagram::sortPersistenceDiagram(
128  vector<tuple<ftm::idVertex,ftm::NodeType,ftm::idVertex,ftm::NodeType,scalarType,ftm::idVertex>>& diagram,
129  scalarType* scalars,
130  int* offsets) const{
131  auto cmp=[scalars, offsets](const tuple<ftm::idVertex,ftm::NodeType,ftm::idVertex,ftm::NodeType,scalarType,ftm::idVertex>& a,
132  const tuple<ftm::idVertex,ftm::NodeType,ftm::idVertex,ftm::NodeType,scalarType,ftm::idVertex>& b){
133  const ftm::idVertex idA=get<0>(a);
134  const ftm::idVertex idB=get<0>(b);
135  const ftm::idVertex va=offsets[idA];
136  const ftm::idVertex vb=offsets[idB];
137  const scalarType sa=scalars[idA];
138  const scalarType sb=scalars[idB];
139 
140  if(sa!=sb) return sa < sb;
141  else return va < vb;
142  };
143 
144  std::sort(diagram.begin(), diagram.end(), cmp);
145 
146  return 0;

149 template <typename scalarType>
150 int PersistenceDiagram::computeCTPersistenceDiagram(
151  ftm::FTMTreePP& tree,
152  const vector<tuple<ftm::idVertex, ftm::idVertex, scalarType, bool>>& pairs,
153  vector<tuple<ftm::idVertex,
154  ftm::NodeType,
155  ftm::idVertex,
156  ftm::NodeType,
157  scalarType,
158  ftm::idVertex>>& diagram,
159  scalarType* scalars) const
160 {
161  const ftm::idVertex numberOfPairs = pairs.size();
162  diagram.resize(numberOfPairs);
163  for (ftm::idVertex i = 0; i < numberOfPairs; ++i) {
164  const ftm::idVertex v0 = get<0>(pairs[i]);
165  const ftm::idVertex v1 = get<1>(pairs[i]);
166  const scalarType persistenceValue = get<2>(pairs[i]);
167  const bool type = get<3>(pairs[i]);
168 
169  get<4>(diagram[i]) = persistenceValue;
170  if (type == true) {
171  get<0>(diagram[i]) = v0;
172  get<1>(diagram[i]) = getNodeType(tree.getJoinTree(), ftm::TreeType::Join, v0);
173  get<2>(diagram[i]) = v1;
174  get<3>(diagram[i]) = getNodeType(tree.getJoinTree(), ftm::TreeType::Join, v1);
175  get<5>(diagram[i]) = 0;
176  } else {
177  get<0>(diagram[i]) = v1;
178  get<1>(diagram[i]) = getNodeType(tree.getSplitTree(), ftm::TreeType::Split, v1);
179  get<2>(diagram[i]) = v0;
180  get<3>(diagram[i]) = getNodeType(tree.getSplitTree(), ftm::TreeType::Split, v0);
181  get<5>(diagram[i]) = 2;
182  }
183  }
184 
185  return 0;
186 }

188 template <typename scalarType>
189 int PersistenceDiagram::execute() const{
190  // get data
191  vector<tuple<ftm::idVertex,ftm::NodeType,ftm::idVertex,ftm::NodeType,scalarType,ftm::idVertex>>& CTDiagram=
192  *static_cast<vector<tuple<ftm::idVertex,ftm::NodeType,ftm::idVertex,ftm::NodeType,scalarType,ftm::idVertex>>*>(CTDiagram_);
193  scalarType* scalars=static_cast<scalarType*>(inputScalars_);
194  int* offsets=static_cast<int*>(inputOffsets_);
195 
196  const ftm::idVertex numberOfVertices=triangulation_->getNumberOfVertices();
197  // convert offsets into a valid format for contour forests
198  vector<ftm::idVertex> voffsets(numberOfVertices);
199  std::copy(offsets,offsets+numberOfVertices,voffsets.begin());
200 
201  // get contour tree
202  ftm::FTMTreePP contourTree;
203  contourTree.setupTriangulation(triangulation_, false);
204  contourTree.setVertexScalars(inputScalars_);
205  contourTree.setTreeType(ftm::TreeType::Join_Split);
206  contourTree.setVertexSoSoffsets(voffsets.data());
207  contourTree.setThreadNumber(threadNumber_);
208  contourTree.setSegmentation(false);
209  contourTree.build<scalarType>();
210 
211  // get persistence pairs
212  vector<tuple<ftm::idVertex, ftm::idVertex, scalarType>> JTPairs;
213  vector<tuple<ftm::idVertex,ftm::idVertex,scalarType>> STPairs;
214  contourTree.computePersistencePairs<scalarType>(JTPairs, true);
215  contourTree.computePersistencePairs<scalarType>(STPairs, false);
216 
217  // merge pairs
218  vector<tuple<ftm::idVertex, ftm::idVertex, scalarType, bool>> CTPairs(JTPairs.size() +
219  STPairs.size());
220  const ftm::idVertex JTSize = JTPairs.size();
221  for (ftm::idVertex i = 0; i < JTSize; ++i) {
222  const auto& x = JTPairs[i];
223  CTPairs[i] = make_tuple(get<0>(x), get<1>(x), get<2>(x), true);
224  }
225  const ftm::idVertex STSize = STPairs.size();
226  for (ftm::idVertex i = 0; i < STSize; ++i) {
227  const auto& x = STPairs[i];
228  CTPairs[JTSize + i] = make_tuple(get<0>(x), get<1>(x), get<2>(x), false);
229  }
230 
231  // remove the last pair which is present two times (global extrema pair)
232  {
233  auto cmp = [](const tuple<ftm::idVertex, ftm::idVertex, scalarType, bool>& a,
234  const tuple<ftm::idVertex, ftm::idVertex, scalarType, bool>& b) {
235  return get<2>(a) < get<2>(b);
236  };
237 
238  std::sort(CTPairs.begin(), CTPairs.end(), cmp);
239  CTPairs.erase(CTPairs.end() - 1);
240  }
241 
242  // get the saddle-saddle pairs
243  vector<tuple<int,int,scalarType>> pl_saddleSaddlePairs;
244  const int dimensionality=triangulation_->getDimensionality();
245  if(dimensionality==3 and ComputeSaddleConnectors){
246  MorseSmaleComplex3D morseSmaleComplex;
247  morseSmaleComplex.setDebugLevel(debugLevel_);
248  morseSmaleComplex.setThreadNumber(threadNumber_);
249  morseSmaleComplex.setupTriangulation(triangulation_);
250  morseSmaleComplex.setInputScalarField(inputScalars_);
251  morseSmaleComplex.setInputOffsets(inputOffsets_);
252  morseSmaleComplex.computePersistencePairs<scalarType>(JTPairs, STPairs, pl_saddleSaddlePairs);
253  }
254 
255  // get persistence diagrams
256  computeCTPersistenceDiagram<scalarType>(
257  contourTree,CTPairs, CTDiagram, scalars);
258 
259  // add saddle-saddle pairs to the diagram if needed
260  if(dimensionality==3 and ComputeSaddleConnectors){
261  for(const auto& i : pl_saddleSaddlePairs){
262  const ftm::idVertex v0=get<0>(i);
263  const ftm::idVertex v1=get<1>(i);
264  const scalarType persistenceValue=get<2>(i);
265 
266  tuple<ftm::idVertex, ftm::NodeType, ftm::idVertex, ftm::NodeType, scalarType, ftm::idVertex> t;
267 
268  get<0>(t)=v0;
269  get<1>(t)=ftm::NodeType::Saddle1;
270  get<2>(t)=v1;
271  get<3>(t)=ftm::NodeType::Saddle2;
272  get<4>(t)=persistenceValue;
273  get<5>(t)=1;
274 
275  CTDiagram.push_back(t);
276  }
277  }
278 
279  // finally sort the diagram
280  sortPersistenceDiagram(CTDiagram, scalars, offsets);
281 
282  return 0;
283 }
