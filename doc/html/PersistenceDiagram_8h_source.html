130 int PersistenceDiagram::computeCTPersistenceDiagram(ContourForests& tree,
131  const vector<tuple<idVertex, idVertex, scalarType, bool>>& pairs,
132  vector<tuple<idVertex,NodeType,idVertex,NodeType,scalarType,idVertex>>& diagram,

136  for(idVertex i=0; i<numberOfPairs; ++i){
137  const idVertex v0=get<0>(pairs[i]);

141 
142  get<4>(diagram[i])=persistenceValue;
143  if(type==true){
144  get<0>(diagram[i])=v0;
145  get<1>(diagram[i])=getNodeType(tree.getJoinTree(),TreeType::Join,v0);
146  get<2>(diagram[i])=v1;
147  get<3>(diagram[i])=getNodeType(tree.getJoinTree(),TreeType::Join,v1);
148  get<5>(diagram[i])=0;
149  }
150  else{
151  get<0>(diagram[i])=v1;
152  get<1>(diagram[i])=getNodeType(tree.getSplitTree(),TreeType::Split,v1);
153  get<2>(diagram[i])=v0;
154  get<3>(diagram[i])=getNodeType(tree.getSplitTree(),TreeType::Split,v0);
155  get<5>(diagram[i])=2;
156  }
157  }

373

375  if(dimensionality==3 and ComputeSaddleConnectors){
376  for(const auto& i : pl_saddleSaddlePairs){
377  const idVertex v0=get<0>(i);

384  get<1>(t)=NodeType::Saddle1;
385  get<2>(t)=v1;
386  get<3>(t)=NodeType::Saddle2;
387  get<4>(t)=persistenceValue;
