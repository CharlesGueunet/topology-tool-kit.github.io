95  template <typename type>
96  void qsort(type arr[], const long int begin, const long int stop,
97  std::function<bool(type, type)> comp) const
98  {
99  if (begin >= stop)
100  return;
101 
102  static const long int MINSIZE = 10;
103 
104  long int left = begin - 1;
105  long int right = stop + 1;
106  const type pivot = arr[begin];
107 
108  while (1) {
109  while (comp(pivot, arr[--right]))
110  ;
111  while (++left <= stop && !comp(pivot, arr[left]))
112  ;
113 
114  if (left < right)
115  swap_el<type>(arr, left, right);
116  else
117  break;
118  }
119 
120  swap_el<type>(arr, begin, right);
121 #pragma omp task untied if (right - begin > MINSIZE)
122  qsort(arr, begin, right - 1, comp);
123 #pragma omp task untied if (stop - right > MINSIZE)
124  qsort(arr, right + 1, stop, comp);
125  }

129  static void swap_el(type arr[], const size_t a, const size_t b)
130  {
131  const type tmp = arr[a];
132  arr[a] = arr[b];
133  arr[b] = tmp;
134  }
