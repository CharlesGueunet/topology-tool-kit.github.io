193 scalars, int* offsets) const{
194  bool isMinima{true};
195  bool isMaxima{true};
196  int neighborNumber=triangulation_->getVertexNeighborNumber(vertex);
197  for(int i=0; i<neighborNumber; ++i){
198  int neighbor;
199  triangulation_->getVertexNeighbor(vertex,i,neighbor);
200 
201  if(isLowerThan<dataType>(neighbor,vertex,scalars,offsets)) isMinima=false;
202  if(isHigherThan<dataType>(neighbor,vertex,scalars,offsets)) isMaxima=false;
203  if(!isMinima and !isMaxima){
204  return 0;
205  }
206  }
207 
208  if(isMinima) return -1;
209  if(isMaxima) return 1;
210 
211  return 0;

257 template <typename dataType>
258 int ttk::TopologicalSimplification::addPerturbation(dataType* scalars, int*
259 offsets) const{
260  dataType epsilon{};
261 
262  if(std::is_same<dataType,double>::value) epsilon=pow10(1-DBL_DIG);
263  else if(std::is_same<dataType,float>::value) epsilon=pow10(1-FLT_DIG);
264  else return -1;
265 
266  std::vector<std::tuple<dataType,int,int>> perturbation(vertexNumber_);
267  for(int i=0; i<vertexNumber_; ++i){
268  std::get<0>(perturbation[i])=scalars[i];
269  std::get<1>(perturbation[i])=offsets[i];
270  std::get<2>(perturbation[i])=i;
271  }
272 
273  SweepCmp cmp(true);
274  sort(perturbation.begin(), perturbation.end(), cmp);
275 
276  for(int i=0; i<vertexNumber_; ++i){
277  if(i){
278  if(std::get<0>(perturbation[i]) <= std::get<0>(perturbation[i-1]))
279  std::get<0>(perturbation[i])=std::get<0>(perturbation[i-1]) + epsilon;
280  }
281  scalars[std::get<2>(perturbation[i])]=std::get<0>(perturbation[i]);
282  }
283 
284  return 0;
285 }

302 #endif

345  // processing

348 
349  {
350  std::stringstream msg;

354  }

356  for(int j=0; j<2; ++j){

359 
360  cmp.setIsIncreasingOrder(isIncreasingOrder);
361  std::set<std::tuple<dataType,int,int>, decltype(cmp)> sweepFront(cmp);
362  std::vector<bool> visitedVertices(vertexNumber_, false);
363  std::vector<int> adjustmentSequence(vertexNumber_);
364 
365  // add the seeds
366  if(isIncreasingOrder){
367  for(int k : authorizedMinima){
368  authorizedExtrema[k]=true;
369  sweepFront.emplace(scalars[k],offsets[k],k);
370  visitedVertices[k]=true;
371  }
372  }
373  else{
374  for(int k : authorizedMaxima){
375  authorizedExtrema[k]=true;
376  sweepFront.emplace(scalars[k],offsets[k],k);
377  visitedVertices[k]=true;
378  }
379  }
380

382  int adjustmentPos = 0;
383  do{
384  auto front=sweepFront.begin();
385  if(front==sweepFront.end()) return -1;
386

388  sweepFront.erase(front);
389

391  for(int k=0; k<neighborNumber; ++k){
392  int neighbor;

394  if(!visitedVertices[neighbor]){

398  }

400  ++adjustmentPos;
401  }while(!sweepFront.empty());

404  int offset = (isIncreasingOrder ? 0 : vertexNumber_ + 1);
405 
406  for(int k=0; k<vertexNumber_; ++k){
407

410  scalars[adjustmentSequence[k]]=scalars[adjustmentSequence[k-1]];

412  }

415  scalars[adjustmentSequence[k]]=scalars[adjustmentSequence[k-1]];

417  }

427

429  if(minima.size() > authorizedMinima.size()) needForMoreIterations=true;
430 
431  {
432  std::stringstream msg;
433  msg << "[TopologicalSimplification] Current status: "
434  << minima.size() << " minima, "
435  << maxima.size() << " maxima." << std::endl;
436  dMsg(std::cout, msg.str(), advancedInfoMsg);
437  }
438 
439  if(!needForMoreIterations){
440  for(int k : minima){
441  if(!authorizedExtrema[k]){
442  needForMoreIterations=true;
443  break;
444  }
445  }
446  }
447  if(!needForMoreIterations){
448  for(int k : maxima){
449  if(!authorizedExtrema[k]){
450  needForMoreIterations=true;
451  break;
452  }
453  }
454  }
455 
456  // optional adding of perturbation
457  if(addPerturbation_) addPerturbation<dataType>(scalars, offsets);
458 
459  ++iteration;
460  if(!needForMoreIterations) break;
461  }
462

464  std::stringstream msg;
465  msg << "[TopologicalSimplification] Scalar field simplified"
466  << " in " << t.getElapsedTime() << " s. (" << threadNumber_
467  << " threads(s), " << iteration << " ite.)."
468  << std::endl;
469  dMsg(std::cout,msg.str(),timeMsg);
470  }
471 
472  return 0;
