153  const auto qCode = codeBuf[q];

156  const bool qIsSad = qCode != minCode && qCode != maxCode;
157  if(pIsSad || qIsSad) {
158  if(pIsSad && qIsSad) // two saddles
159  continue;
160  // extremum and saddle
161  const auto ext = pIsSad ? q : p;
162  const auto sad = pIsSad ? p : q;
163  if(sizeBuf[ext] < minSize)
164  continue;
165  _isovals.push_back(scalarBuf[ext] * extFac + scalarBuf[sad] * sadFac);
166  const auto point = &coords[ext * 3];
167  _coords.push_back(point[0]);
168  _coords.push_back(point[1]);
169  _coords.push_back(point[2]);
170  _flags.push_back(codeBuf[ext] == minCode ? 0 : 1);
171  } else // min-max pair

174  const auto qVal = scalarBuf[q];

176  const auto cVal
177  = (pVal + qVal) / 2;
178  if(sizeBuf[p] >= minSize) {
179  _isovals.push_back(pVal * extFac + cVal * sadFac);
180  const auto point = &coords[p * 3];
181  _coords.push_back(point[0]);
182  _coords.push_back(point[1]);
183  _coords.push_back(point[2]);
184  _flags.push_back(pCode == minCode ? 0 : 1);
185  }

187  _isovals.push_back(qVal * extFac + cVal * sadFac);
188  const auto point = &coords[q * 3];
189  _coords.push_back(point[0]);
190  _coords.push_back(point[1]);
