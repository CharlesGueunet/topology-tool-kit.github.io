137 void FTMTree_MT::arcGrowth(const idVertex startVert, const idVertex orig)
138 {
139  // current task id / propag
140 
141  // local order (ignore non regular verts)
142  idVertex localOrder = -1;
143  UF startUF = (*mt_data_.ufs)[startVert]->find();
144  // get or recover states
145  CurrentState *currentState;
146  if (startUF->getNbStates()) {
147  currentState = startUF->getFirstState();
148  } else {
149  const std::size_t currentStateId = mt_data_.states->getNext();
150  currentState = &(*mt_data_.states)[currentStateId];
151  currentState->setStartVert(startVert);
152  startUF->addState(currentState);
153  }
154 
155  currentState->addNewVertex(startVert);
156 
157  // avoid duplicate processing of startVert
158  bool seenFirst = false;
159 
160  // ARC OPENING
161  idNode startNode = getCorrespondingNodeId(startVert);
162  idSuperArc currentArc = openSuperArc(startNode);
163  startUF->addArcToClose(currentArc);
164 #ifdef TTK_ENABLE_FTM_TREE_STATS_TIME
165  (*mt_data_.activeTasksStats)[currentArc].begin = _launchGlobalTime.getElapsedTime();
166  (*mt_data_.activeTasksStats)[currentArc].origin = orig;
167 #endif

171  // Next vertex
172 
173  idVertex currentVert = currentState->getNextMinVertex();
174 
175  // ignore duplicate
176  if (!isCorrespondingNull(currentVert) && !isCorrespondingNode(currentVert)) {
177  continue;
178  } else {
179  // first node can be duplicate, avoid duplicate process
180  if (currentVert == startVert) {
181  if (!seenFirst) {
182  seenFirst = true;
183  } else {
184  continue;

186  }

925  }
926

928  };

931  if (goUp) {
932  return getSuperArc(curArc)->getUpNodeId();
933  }
934 
935  return getSuperArc(curArc)->getDownNodeId();
936  };
937 
938  std::queue<tuple<idNode, bool>> q;
939  std::stack<tuple<idNode, bool>> qr;
940  for (const idNode n : *mt_data_.leaves) {
941  bool goUp = isJT() || isST() || getNode(n)->getNumberOfUpSuperArcs();
942  if (goUp)
943  q.emplace(make_tuple(n, goUp));
944  else
945  qr.emplace(make_tuple(n, goUp));
946  }
947 
948  while (!qr.empty()) {
949  q.emplace(qr.top());
950  qr.pop();

955  idSuperArc nIdMax = getNumberOfSuperArcs() - 1;
956 
957  vector<bool> seenUp(getNumberOfSuperArcs(), false);
958  vector<bool> seenDown(getNumberOfSuperArcs(), false);
959

961  bool goUp;
962  idNode curNodeId;
963  tie(curNodeId, goUp) = q.front();
964  q.pop();

977  getSuperArc(currentArcId)->setNormalizeIds(nIdMin++);
978  }

986  }
987  if (!seenDown[currentArcId]) {
988  q.emplace(make_tuple(getArcParentNode(currentArcId, goUp), goUp));

990  }
991  }
992  }
993  }
