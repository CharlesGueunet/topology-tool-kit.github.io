208 template<typename scalarT>
209 int ttk::ContourAroundPoint::compOneContour(SimplexId cCenter, float isovalue, int flag) const
210 {
211  const SimplexId nc = _inpFieldTriangulation->getNumberOfCells();
212  // With a DFS starting at cell `cCenter` we find the first cell that contains the isovalue
213  // (one vertex is below, one above the isovalue), call it `cStart`.
214  // From `cStart` we extend the isocontour by searching for all neighboring cells that also contain
215  // the isovalue. The difference to the first search is that this time we don't enqueue the
216  // neighbors of cells that don't contain the isovalue.
217  // (But we can still use the same `visited` set.)
218  // NOTE This approach assumes that the isocontour is a single connected component.
219  std::set<SimplexId> visited;
220 
221  // Find an initial cell which contais the isovalue.
222  SimplexId cStart = nc;
223  auto q = std::stack<SimplexId>({cCenter}); // LIFO queue for DFS
224  while(!q.empty())
225  {
226  const auto c = q.top(); q.pop();
227  visited.insert(c);
228  if(checkContains<scalarT>(c, isovalue)) {
229  cStart = c;
230  break; // We don't have to search further, we have a cell that contains the isovalue.
231  }
232  enqueueNeighbors(c, q, visited);
233  }
234  if(cStart == nc)
235  return -1;
236 
237  // Compute the cells contributing to the contour starting at the initial cell.
238  auto isocells = std::vector<SimplexId>{cStart};
239  q = std::stack<SimplexId>(); // reset q
240  enqueueNeighbors(cStart, q, visited);
241  while(!q.empty())
242  {
243  const auto c = q.top(); q.pop();
244  visited.insert(c);
245  if(checkContains<scalarT>(c, isovalue)) {
246  isocells.push_back(c);
247  enqueueNeighbors(c, q, visited);
248  }
249  }
250 #if 0
251  addRoughOutput<scalarT>(isocells);
252 #else
253  addOutput<scalarT>(isocells, isovalue, flag);
254 #endif
255  if(isocells.size() == 1)
256  return -2;
257 
258  return 0;
259 }

331 template<typename scalarT>
332 void ttk::ContourAroundPoint::addOutput(const std::vector<SimplexId>& cellIds, float isoval,
333  int flag) const
334 {
335  _outNvPerC = _inpNvPerC - 1;
336  const SimplexId inpNePerC = _inpNvPerC == 3 ? 3 : 6;
337  std::map<SimplexId, SimplexId> e2v;
338  // NOTE We simply create line segments when it would be better to create line strips.
339 
340  auto inpScalars = reinterpret_cast<const scalarT*>(_inpFieldScalars);
341 
342  for(const auto c : cellIds)
343  {
344  for(SimplexId i = 0; i < inpNePerC; ++i)
345  {
346  SimplexId e; _inpFieldTriangulation->getCellEdge(c, i, e);
347  if(e2v.count(e) != 0) {
348  _outFieldCinfos.push_back(e2v[e]);
349  continue;
350  }
351 
352  SimplexId p; _inpFieldTriangulation->getEdgeVertex(e, 0, p);
353  SimplexId q; _inpFieldTriangulation->getEdgeVertex(e, 1, q);
354  const float pVal = static_cast<float>(inpScalars[p]);
355  const float qVal = static_cast<float>(inpScalars[q]);
356  if((pVal < isoval && qVal > isoval) || (pVal > isoval && qVal < isoval))
357  {
358  const SimplexId v = _outFieldScalars.size();
359  e2v[e] = v;
360  float px, py, pz; _inpFieldTriangulation->getVertexPoint(p, px, py, pz);
361  float qx, qy, qz; _inpFieldTriangulation->getVertexPoint(q, qx, qy, qz);
362  const double pFac = (qVal - isoval) / (qVal - pVal);
363  const double qFac = 1 - pFac;
364  const float x = px * pFac + qx * qFac;
365  const float y = py * pFac + qy * qFac;
366  const float z = pz * pFac + qz * qFac;
367  _outFieldCoords.push_back(x);
368  _outFieldCoords.push_back(y);
369  _outFieldCoords.push_back(z);
370  _outFieldCinfos.push_back(v);
371  _outFieldScalars.push_back(isoval);
372  _outFieldFlags.push_back(flag);
373  }
374  }
375  }
