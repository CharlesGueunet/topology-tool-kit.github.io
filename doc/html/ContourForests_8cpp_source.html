444 
445  // Get the good tree
446  auto getTreePart = [&, treetype](const idPartition &i) -> MergeTree * {
447  if (treetype == 0) {
448  return parallelData_.trees[i].getJoinTree();
449  }
450  if (treetype == 1) {
451  return parallelData_.trees[i].getSplitTree();
452  }
453  return &parallelData_.trees[i];
454  };
455 
456  // this tree will receive the final tree
457  // all variables linked to tmpree have a "_tt" suffix
458  MergeTree tmpTree(params_,mesh_, scalars_, params_->treeType);
459  // for vert2tree
460  tmpTree.flush();
461  // statistical reserves
462  tmpTree.treeData_.nodes.reserve(scalars_->size / 50);
463  tmpTree.treeData_.superArcs.reserve(scalars_->size / 50);
464 
465  // partition, node in partion, is a leaf
466  queue<tuple<idInterface, idNode>> leavesNodes;
467  vector<unsigned> nbVisit(scalars_->size, 0);
468 
469  // Unify by traversing leaves for each partition starting by the lowest
470  for (idPartition partition = 0; partition < parallelParams_.nbPartitions; ++partition) {
471  MergeTree *currentTree = getTreePart(partition);
472 
473  for (auto &l : currentTree->treeData_.leaves) {
474  Node * curNode = currentTree->getNode(l);
475  idVertex leafVert = curNode->getVertexId();
476 
477  // Condition to keep
478 
479  // if not in partition
480  if (partition != 0 && isLower(leafVert, parallelData_.interfaces[partition - 1].getSeed()))
481  continue;
482  if (partition != parallelParams_.nbInterfaces &&
483  isHigher(leafVert, parallelData_.interfaces[partition].getSeed()))
484  continue;
485 
486  // if hidden
487  if (!curNode->isVisible())
488  continue;
489 
490  // if max
491  if (!curNode->getNumberOfUpSuperArcs())
492  continue;
493 
494  // this leaf have received an external arc during stitching, not a real leaf anymore
495  if (curNode->getNumberOfUpSuperArcs() &&
496  curNode->getNumberOfDownSuperArcs())
497  continue;
498 
499  // isolated node
500  if (!currentTree->getNumberOfVisibleArcs(l))
501  continue;
502 
503  // Add the leave
504 
505  if (!nbVisit[leafVert]) {
506  // will be processed
507  leavesNodes.emplace(partition, l);
508 
509  if (DEBUG) {
510  cout << "will see : partition : " << static_cast<unsigned>(partition);
511  cout << " leaf node " << currentTree->printNode(l) << endl;
512  }
513 
514  // seen once
515  ++nbVisit[leafVert];
516 
517  // Create node in tmpTree and mark as leaf
518  const idNode &newNodeId_tt = tmpTree.makeNode(curNode);
519  tmpTree.treeData_.leaves.emplace_back(newNodeId_tt);
520  }
521  }
522  }
523 
524 
525  // cross node from min to max to construct the tree
526 
527  while (!leavesNodes.empty()) {
528  // get the next node
529  idPartition currentPartition;
530  idNode currentNodeId;
531  tie(currentPartition, currentNodeId) = leavesNodes.front();
532  leavesNodes.pop();
533 
534  // get tree and node
535  MergeTree *currentTree = getTreePart(currentPartition);
536  Node * currentNode = currentTree->getNode(currentNodeId);
537 
538  if (DEBUG) {
539  cout << endl;
540  cout << "process : partition : " << static_cast<unsigned>(currentPartition) << endl;
541  cout << " node " << currentTree->printNode(currentNodeId) << endl;
542  }
543 
544  // create or recover in tmpTree
545  const idNode &baseNode_tt = tmpTree.makeNode(currentNode);
546 
547  // Cross the ups arc of the node
548  const idPartition refPartition = currentPartition;
549  const idNode refNodeId = currentNodeId;
550  const idSuperArc &nbUpArc = currentNode->getNumberOfUpSuperArcs();
551 
552  for (idSuperArc upArcPos = 0; upArcPos < nbUpArc; upArcPos++) {
553  // reset partition / node and tree

556  currentNodeId = refNodeId;
557  currentNode = currentTree->getNode(currentNodeId);
558

561 
562  if(DEBUG){
563  cout << " process arc " << currentTree->printArc(upArcId) << endl;
564  }
565
