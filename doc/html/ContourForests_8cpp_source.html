448 void ContourForests::unifyTree(const char treetype)
449 {
450 
451  const bool DEBUG = false;
452 
453  // Get the good tree
454  auto getTreePart = [&, treetype](const idPartition &i) -> MergeTree * {
455  if (treetype == 0) {
456  return parallelData_.trees[i].getJoinTree();
457  }
458  if (treetype == 1) {
459  return parallelData_.trees[i].getSplitTree();
460  }
461  return &parallelData_.trees[i];
462  };
463 
464  // this tree will receive the final tree
465  // all variables linked to tmpree have a "_tt" suffix
466  MergeTree tmpTree(params_,mesh_, scalars_, params_->treeType);
467  // for vert2tree
468  tmpTree.flush();
469  // statistical reserves
470  tmpTree.treeData_.nodes.reserve(scalars_->size / 50);
471  tmpTree.treeData_.superArcs.reserve(scalars_->size / 50);
472 
473  // partition, node in partion, is a leaf
474  queue<tuple<idInterface, idNode>> leavesNodes;
475  vector<unsigned> nbVisit(scalars_->size, 0);
476 
477  // Unify by traversing leaves for each partition starting by the lowest
478  for (idPartition partition = 0; partition < parallelParams_.nbPartitions; ++partition) {
479  MergeTree *currentTree = getTreePart(partition);
480 
481  for (auto &l : currentTree->treeData_.leaves) {
482  Node * curNode = currentTree->getNode(l);
483  SimplexId leafVert = curNode->getVertexId();
484 
485  // Condition to keep
486 
487  // if not in partition
488  if (partition != 0 && isLower(leafVert, parallelData_.interfaces[partition - 1].getSeed()))
489  continue;
490  if (partition != parallelParams_.nbInterfaces &&
491  isHigher(leafVert, parallelData_.interfaces[partition].getSeed()))
492  continue;
493 
494  // if hidden
495  if (!curNode->isVisible())
496  continue;
497 
498  // if max
499  if (!curNode->getNumberOfUpSuperArcs())
500  continue;
501 
502  // this leaf have received an external arc during stitching, not a real leaf anymore
503  if (curNode->getNumberOfUpSuperArcs() &&
504  curNode->getNumberOfDownSuperArcs())
505  continue;
506 
507  // isolated node
508  if (!currentTree->getNumberOfVisibleArcs(l))
509  continue;
510 
511  // Add the leave
512 
513  if (!nbVisit[leafVert]) {
514  // will be processed
515  leavesNodes.emplace(partition, l);
516 
517  if (DEBUG) {
518  cout << "will see : partition : " << static_cast<unsigned>(partition);
519  cout << " leaf node " << currentTree->printNode(l) << endl;
520  }
521 
522  // seen once
523  ++nbVisit[leafVert];
524 
525  // Create node in tmpTree and mark as leaf
526  const idNode &newNodeId_tt = tmpTree.makeNode(curNode);
527  tmpTree.treeData_.leaves.emplace_back(newNodeId_tt);
528  }
529  }
530  }
531 
532 
533  // cross node from min to max to construct the tree
534 
535  while (!leavesNodes.empty()) {
536  // get the next node
537  idPartition currentPartition;
538  idNode currentNodeId;
539  tie(currentPartition, currentNodeId) = leavesNodes.front();
540  leavesNodes.pop();
541 
542  // get tree and node
543  MergeTree *currentTree = getTreePart(currentPartition);
544  Node * currentNode = currentTree->getNode(currentNodeId);
545 
546  if (DEBUG) {
547  cout << endl;
548  cout << "process : partition : " << static_cast<unsigned>(currentPartition) << endl;
549  cout << " node " << currentTree->printNode(currentNodeId) << endl;
550  }
551 
552  // create or recover in tmpTree
553  const idNode &baseNode_tt = tmpTree.makeNode(currentNode);
554 
555  // Cross the ups arc of the node
556  const idPartition refPartition = currentPartition;
557  const idNode refNodeId = currentNodeId;
558  const idSuperArc &nbUpArc = currentNode->getNumberOfUpSuperArcs();
559 
560  for (idSuperArc upArcPos = 0; upArcPos < nbUpArc; upArcPos++) {
561  // reset partition / node and tree
562  currentPartition = refPartition;
563  currentTree = getTreePart(refPartition);
564  currentNodeId = refNodeId;
565  currentNode = currentTree->getNode(currentNodeId);
566 
567  const idSuperArc & upArcId = currentNode->getUpSuperArcId(upArcPos);
568  SuperArc * upArc = currentTree->getSuperArc(upArcId);
569 
570  if(DEBUG){
571  cout << " process arc " << currentTree->printArc(upArcId) << endl;
572  }
573 
574  if (!upArc->isVisible()) {
575  if (DEBUG) {
576  cout << " - ignore not visible" << endl;
577  }
578  continue;
579  }
580 
581  const idSuperArc & newArcId_tt = tmpTree.openSuperArc(baseNode_tt, false, false);
582  SuperArc * newArc_tt = tmpTree.getSuperArc(newArcId_tt);
583 
584  // segmentation related
585  list<pair<SimplexId, bool> *> listVertList;
586  list<SimplexId> listVertSize;
587  int totalSize=0;
588 
589  // add upArc until the upnode is not a regular one
590  do{
591  listVertList.push_back(upArc->getVertList());
592  listVertSize.push_back(upArc->getVertSize());
593  totalSize += upArc->getVertSize();
594 
595  upArc->hide();
596 
597  currentPartition = upArc->getUpCT();
598  currentTree = getTreePart(currentPartition);
599  currentNodeId = upArc->getUpNodeId();
600  currentNode = currentTree->getNode(currentNodeId);
601 
602  if (currentNode->getNumberOfUpSuperArcs() == 1 &&
603  currentNode->getNumberOfDownSuperArcs() == 1) {
604  upArc = currentTree->getSuperArc(currentNode->getUpSuperArcId(0));
605  if (DEBUG) {
606  cout << " cross : " << currentTree->printArc(currentNode->getUpSuperArcId(0));
607  cout << endl;
608  }
609  } else {
610  // no longer regular : stop here
611  if(DEBUG){
612  cout << "stop at " << currentTree->printNode(currentNodeId) << endl;
613  }
614  break;
615  }
616 
617  } while (true);
618 
619  // Finish the current Arc (segmentation + close)
620  if(totalSize){
621  newArc_tt->appendVertLists(listVertList, listVertSize, totalSize);
622  }
623  const idNode &closingNode_tt = tmpTree.makeNode(currentNode);
624  tmpTree.closeSuperArc(newArcId_tt, closingNode_tt, false, false);
625 
626  if (DEBUG) {
627  cout << " Create arc : " << tmpTree.printArc(newArcId_tt) << endl;
628  }
629 
630  // push current vertex TODO
631  const SimplexId &closingVertex = currentNode->getVertexId();
632  ++nbVisit[closingVertex];
633 
634  // get the down valence of this node in the partition where it have no noise
635  const idPartition &closingPartition = vertex2partition(closingVertex);
636  MergeTree * closingTree = getTreePart(closingPartition);
637  const idNode & closingNodeId = closingTree->getCorrespondingNodeId(closingVertex);
638  const unsigned &downVal = closingTree->getNode(closingNodeId)->getNumberOfDownSuperArcs();
639 
640  if (nbVisit[closingVertex] == downVal) {
641  leavesNodes.emplace(closingPartition, closingNodeId);
642  if (DEBUG) {
643  cout << " push : partition : " << static_cast<unsigned>(closingPartition) << endl;
644  cout << " push : node : " << closingTree->printNode(closingNodeId) << endl;
645  }
646  } else if (DEBUG){
647  cout << " visit : " << nbVisit[closingVertex] << endl;
648  cout << " downVal : " << downVal << endl;
649  }
650  } // end for each up arc
651  } // end while leavesNodes
652 
653  tmpTree.treeData_.superArcs.shrink_to_fit();
654  tmpTree.treeData_.nodes.shrink_to_fit();
655

658  jt_->clone(&tmpTree);
659  } else if (treetype == 1) {
