468  {

474  << std::endl;

776 template <class dataTypeU, class dataTypeV>
777  inline int ttk::ReebSpace::computeGeometricalMeasures(Sheet3 &sheet){
778 
779  sheet.domainVolume_ = 0;
780  sheet.rangeArea_ = 0;
781  sheet.hyperVolume_ = 0;
782 
783  for(SimplexId i = 0; i < (SimplexId) sheet.tetList_.size(); i++){
784 
785  SimplexId tetId = sheet.tetList_[i];
786  std::vector<std::pair<double, double> > domainBox, rangeBox;
787  std::vector<std::vector<float> > domainPoints(4), rangePoints(4);
788 
789  for(int j = 0; j < 4; j++){
790  domainPoints[j].resize(3);
791  rangePoints[j].resize(2);
792 
793  SimplexId vertexId = -1;
794  triangulation_->getCellVertex(tetId, j, vertexId);
795 
796  triangulation_->getVertexPoint(vertexId,
797  domainPoints[j][0], domainPoints[j][1], domainPoints[j][2]);
798 
799  rangePoints[j][0] = ((dataTypeU *) uField_)[vertexId];
800  rangePoints[j][1] = ((dataTypeV *) vField_)[vertexId];
801  }
802 
803  Geometry::getBoundingBox(domainPoints, domainBox);
804  Geometry::getBoundingBox(rangePoints, rangeBox);
805 
806  sheet.domainVolume_ +=
807  (domainBox[0].second - domainBox[0].first)
808  *(domainBox[1].second - domainBox[1].first)
809  *(domainBox[2].second - domainBox[2].first);
810 
811  sheet.rangeArea_ +=
812  (rangeBox[0].second - rangeBox[0].first)
813  *(rangeBox[1].second - rangeBox[1].first);
814 
815  }
816 
817  if(sheet.domainVolume_){
818  sheet.hyperVolume_ = sheet.rangeArea_/sheet.domainVolume_;
819  }
820  else{
821  sheet.hyperVolume_ = 0;
822  }
823 
824  return 0;
825 }
