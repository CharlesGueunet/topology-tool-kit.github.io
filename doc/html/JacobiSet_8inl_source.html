719  break;
720  }
721  }
722  }
723  }
724  }

726  // update the UF if necessary
727  for(SimplexId i = 0; i < (SimplexId)lowerList.size(); i++)
728  lowerList[i] = lowerList[i]->find();
729  for(SimplexId i = 0; i < (SimplexId)upperList.size(); i++)
730  upperList[i] = upperList[i]->find();

733  sort(lowerList.begin(), lowerList.end());
734  it = unique(lowerList.begin(), lowerList.end());
735  lowerList.resize(distance(lowerList.begin(), it));
736 
737  sort(upperList.begin(), upperList.end());
738  it = unique(upperList.begin(), upperList.end());
739  upperList.resize(distance(upperList.begin(), it));
740 
741  if((upperList.size() == 1) && (lowerList.size() == 1))
742  return -2;
743 
744  return 1;
