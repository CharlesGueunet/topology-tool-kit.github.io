261  const int nbRow,
262  const int nbCol,
263  const dataType **matrix) const
264 {
265  dataType col, row;
266  vector<dataType> rowCols(0);
267  for (int i = 0; i < nbRow; ++i) {

319 template <typename dataType>
320 void BottleneckDistance::iterateSolving(
321  vector<matchingTuple>* matchings,
322  const dataType threshold,
323  const int nbRow,
324  const int nbCol,
325  const dataType **matrix,
326  dataType **bottleneckMatrix,
327  Munkres *solver)
328 {
329  // Progressively solve easier affectation problems until
330  // a non-infinite cost is found.
331 
332  dataType adaptiveThreshold = threshold;
333 
334  // Check assignment maximum (infty => restart Munkres with a higher - next - threshold).
335  int minIter = 2;
336  while (!this->isValidMatching(const_cast<const vector<matchingTuple>* >(matchings), adaptiveThreshold)
337  && minIter <= nbRow * nbCol)
338  {
339  matchings->clear();
340 
341  // Find next threshold.
342  dataType newThreshold = numeric_limits<dataType>::max();
343  for (int i = 0; i < nbRow; ++i) {
344  for (int j = 0; j < nbCol; ++j) {
345  dataType e = matrix[i][j];
346  if (e > adaptiveThreshold && e < newThreshold) newThreshold = e;
347  }
348  }
349  adaptiveThreshold = newThreshold;
350 
351  // Apply Munkres again.
352  for (int i = 0; i < nbRow; ++i) {
353  for (int j = 0; j < nbCol; ++j) {
354  dataType e = matrix[i][j];
355  bottleneckMatrix[i][j] =
356  (e > adaptiveThreshold) ? numeric_limits<dataType>::max() : e;
357  }
358  }
359 
360  solver->run<dataType>(matchings);
361 
362  minIter++;
363  }
364 }

384  vector<matchingTuple> *matchings,
385  Munkres *solver)
386 {
387  // Cf. "A New Algorithm for Solving Linear Bottleneck Assignment Problem"
388  // P. S. Pundir, S. K. Porwal, B. P. Singh
389  // Journal of Institute of Science and Technology.
390 
391  // Filter values.

407  solver->run<dataType>(matchings);
