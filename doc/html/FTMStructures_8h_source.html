81  template <typename type>
82  void qsort(type arr[],
83  const long int begin,
84  const long int stop,
85  std::function<bool(type, type)> comp) const {
86  if(begin >= stop)
87  return;
88 
89 #ifdef TTK_ENABLE_OPENMP
90  static const long int MINSIZE = 10;
91 #endif
92 
93  long int left = begin - 1;
94  long int right = stop + 1;
95  const type pivot = arr[begin];
96 
97  while(1) {
98  while(comp(pivot, arr[--right]))
99  ;
100  while(++left <= stop && !comp(pivot, arr[left]))
101  ;
102 
103  if(left < right)
104  swap_el<type>(arr, left, right);
105  else
106  break;
107  }
108 
109  swap_el<type>(arr, begin, right);
110 #ifdef TTK_ENABLE_OPENMP
111 #pragma omp task untied if(right - begin > MINSIZE)
112 #endif
113  qsort(arr, begin, right - 1, comp);
114 #ifdef TTK_ENABLE_OPENMP
115 #pragma omp task untied if(stop - right > MINSIZE)
116 #endif
117  qsort(arr, right + 1, stop, comp);
118  }
