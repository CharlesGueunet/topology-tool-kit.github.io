42  inline static ttk::CriticalType getNodeType(ttk::ftm::FTMTree_MT& tree,
43  const ttk::ftm::idNode nodeId,
44  ttk::ftm::Params params)
45  {
46  const ttk::ftm::Node* node = tree.getNode(nodeId);
47  int upDegree{};
48  int downDegree{};
49  if (params.treeType == ttk::ftm::TreeType::Join or
50  params.treeType == ttk::ftm::TreeType::Contour) {
51  upDegree = node->getNumberOfUpSuperArcs();
52  downDegree = node->getNumberOfDownSuperArcs();
53  } else {
54  downDegree = node->getNumberOfUpSuperArcs();
55  upDegree = node->getNumberOfDownSuperArcs();
56  }
57  int degree = upDegree + downDegree;
58 
59  // saddle point
60  if (degree > 1) {
61  if (upDegree == 2 and downDegree == 1)
62  return ttk::CriticalType::Saddle2;
63  else if (upDegree == 1 and downDegree == 2)
64  return ttk::CriticalType::Saddle1;
65  else if (upDegree == 1 and downDegree == 1)
66  return ttk::CriticalType::Regular;
67  else
68  return ttk::CriticalType::Degenerate;
69  }
70  // local extremum
71  else {
72  if (upDegree)
73  return ttk::CriticalType::Local_minimum;
74  else
75  return ttk::CriticalType::Local_maximum;
76  }
77  }
