40  inline static ftm::NodeType getNodeType(ftm::FTMTree_MT& tree, const ftm::idNode nodeId,
41  ftm::Params params)
42  {
43  const ftm::Node* node = tree.getNode(nodeId);
44  int upDegree{};
45  int downDegree{};
46  if (params.treeType == ftm::TreeType::Join or params.treeType == ftm::TreeType::Contour) {
47  upDegree = node->getNumberOfUpSuperArcs();
48  downDegree = node->getNumberOfDownSuperArcs();
49  } else {
50  downDegree = node->getNumberOfUpSuperArcs();
51  upDegree = node->getNumberOfDownSuperArcs();
52  }
53  int degree = upDegree + downDegree;
54 
55  // saddle point
56  if (degree > 1) {
57  if (upDegree == 2 and downDegree == 1)
58  return ftm::NodeType::Saddle2;
59  else if (upDegree == 1 and downDegree == 2)
60  return ftm::NodeType::Saddle1;
61  else if (upDegree == 1 and downDegree == 1)
62  return ftm::NodeType::Regular;
63  else
64  return ftm::NodeType::Degenerate;
65  }
66  // local extremum
67  else {
68  if (upDegree)
69  return ftm::NodeType::Local_minimum;
70  else
71  return ftm::NodeType::Local_maximum;
72  }

353

355  const double regionSpan = Geometry::distance(coordUp, coordDown);
356 
357  ftm::idSuperArc nid = arc->getNormalizedId();
358 
359  ftm::ArcType regionType;
360  // RegionType
361  if (upNodeType == ftm::NodeType::Local_minimum && downNodeType == ftm::NodeType::Local_maximum)
362  regionType = ftm::ArcType::Min_arc;
363  else if (upNodeType == ftm::NodeType::Local_minimum || downNodeType == ftm::NodeType::Local_minimum)
364  regionType = ftm::ArcType::Min_arc;
365  else if (upNodeType == ftm::NodeType::Local_maximum || downNodeType == ftm::NodeType::Local_maximum)
366  regionType = ftm::ArcType::Max_arc;
367  else if (upNodeType == ftm::NodeType::Saddle1 && downNodeType == ftm::NodeType::Saddle1)
368  regionType = ftm::ArcType::Saddle1_arc;
369  else if (upNodeType == ftm::NodeType::Saddle2 && downNodeType == ftm::NodeType::Saddle2)
370  regionType = ftm::ArcType::Saddle2_arc;
371  else

375 
376  // critical points
377  if(params.normalize){
378  ids->SetTuple1(g_upVertexId , idOffset + nid);
379  ids->SetTuple1(g_downVertexId , idOffset + nid);
380  } else{
381  ids->SetTuple1(g_upVertexId , idOffset + arcId);
382  ids->SetTuple1(g_downVertexId , idOffset + arcId);
383  }
384
