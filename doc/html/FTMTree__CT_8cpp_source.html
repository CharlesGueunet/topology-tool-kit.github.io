344 void FTMTree_CT::createCTArcSegmentation(idSuperArc ctArc, const bool isJT, idSuperArc xtArc)
345 {
346  const FTMTree_MT *xt = (isJT) ? jt_ : st_;
347 
348  /*Here we prefere to create lots of small region, each arc having its own segmentation with
349  * no overlap instead of having a same vertice in several arc and using vert2tree to decide
350  * because we do not want to maintain vert2tree information during the whole computation*/
351  const list<Region> &xtRegions = xt->getSuperArc(xtArc)->getRegions();
352  for (const Region &reg : xtRegions) {
353  segm_it cur = reg.segmentBegin;
354  segm_it end = reg.segmentEnd;
355  segm_it tmpBeg = reg.segmentBegin;
356  // each element inside this region
357  for (; cur != end; ++cur) {
358  if (isCorrespondingNull(*cur)) {
359  updateCorrespondingArc(*cur, ctArc);
360  } else {
361  // already set, we finish a region
362  if (cur != tmpBeg) {
363  getSuperArc(ctArc)->concat(tmpBeg, cur);
364  }
365  // if several contiguous vertices are discarded
366  // cur will be equals to tmpBeg and we will not create empty regions
367  tmpBeg = cur + 1;
368  }
369  }
370  // close last region
371  if (cur != tmpBeg) {
372  getSuperArc(ctArc)->concat(tmpBeg, cur);
373  }
374  }

416 int FTMTree_CT::leafSearch()
417 {
418  const auto nbScalars = scalars_->size;
419  const auto chunkSize = getChunkSize();
420  const auto chunkNb = getChunkCount();
421 
422  // Extrema extract and launch tasks
423  for (idVertex chunkId = 0; chunkId < chunkNb; ++chunkId) {
424 #ifdef TTK_ENABLE_OPENMP
425 #pragma omp task firstprivate(chunkId)
426 #endif
427  {
428  const idVertex lowerBound = chunkId * chunkSize;
429  const idVertex upperBound = min(nbScalars, (chunkId + 1) * chunkSize);
430  for (idVertex v = lowerBound; v < upperBound; ++v) {
431  const auto &neighNumb = mesh_->getVertexNeighborNumber(v);
432  valence upval = 0;
433  valence downval = 0;
434 
435  for (valence n = 0; n < neighNumb; ++n) {
436  idVertex neigh;
437  mesh_->getVertexNeighbor(v, n, neigh);
438  if(scalars_->isLower(neigh, v)){
439  ++downval;
440  } else {
441  ++upval;
442  }
443  }
444 
445  jt_->setValence(v, downval);
446  st_->setValence(v, upval);
447 
448  if (!downval) {
449  jt_->makeNode(v);
450  }
451 
452  if (!upval) {
453  st_->makeNode(v);
454  }
455  }
456  }
457  }
458 
459 #ifdef TTK_ENABLE_OPENMP
460 #pragma omp taskwait
461 #endif
462  return 0;
