103 template <typename scalarType>
104 int PersistenceCurve::computePersistencePlot(const vector<tuple<ftm::idVertex, ftm::idVertex, scalarType>>& pairs,
105  vector<pair<scalarType, ftm::idVertex>> &plot) const{
106 
107  ftm::idVertex nbElmnt = pairs.size();
108  plot.resize(nbElmnt);
109 
110  // build curve
111  const scalarType epsilon=static_cast<scalarType>(pow(10, -REAL_SIGNIFICANT_DIGITS));
112  for (ftm::idVertex i = 0; i < nbElmnt; ++i) {
113  plot[i].first = std::max(get<2>(pairs[i]), epsilon);
114  plot[i].second = pairs.size() - i;
115  }
116 
117  return 0;
118 }

152  std::copy(JTPairs.begin(), JTPairs.end(), CTPairs.begin());
153  std::copy(STPairs.begin(), STPairs.end(), CTPairs.begin() + JTPairs.size());

160  }

184  }
