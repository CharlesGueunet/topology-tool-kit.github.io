107 template <typename scalarType>
108 int ttk::PersistenceCurve::computePersistencePlot(const
109 std::vector<std::tuple<SimplexId, SimplexId, scalarType>>& pairs,
110  std::vector<std::pair<scalarType, SimplexId>> &plot) const{
111 
112  SimplexId nbElmnt = pairs.size();
113  plot.resize(nbElmnt);
114 
115  // build curve
116  const scalarType epsilon=static_cast<scalarType>(pow(10,
117 -REAL_SIGNIFICANT_DIGITS));
118  for (SimplexId i = 0; i < nbElmnt; ++i) {
119  plot[i].first = std::max(std::get<2>(pairs[i]), epsilon);
120  plot[i].second = pairs.size() - i;
121  }
122 
123  return 0;
124 }
