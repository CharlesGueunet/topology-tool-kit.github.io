384 bool RangeDrivenOctree::segmentIntersection(
385  const pair<double, double> &p0,
386  const pair<double, double> &p1,
387  const pair<double, double> &q0,
388  const pair<double, double> &q1) const{
389 
390  // NOTE: [q0, q1] is axis aligned
391  bool horizontal = true;
392  double x = 0, y = 0;
393 
394  if(q0.first == q1.first) horizontal = false;
395 
396  double denP = p1.first - p0.first;
397  if(!denP) denP = DBL_EPSILON;
398 
399  double P = (p1.second - p0.second)/denP;
400  if(!P) P = DBL_EPSILON;
401  double bP = p1.second - P*p1.first;
402 
403  if(horizontal){
404  y = q0.second;
405  x = (y - bP)/P;
406  }
407  else{
408  x = q0.first;
409  y = P*x + bP;
410  }
411 
412  // does the intersection lands on the range segment?
413  if(horizontal){
414  if((x < fmin(q0.first, q1.first))||(x > fmax(q0.first, q1.first))){
415  return false;
416  }
417  }
418  else{
419  if((y < fmin(q0.second, q1.second))||(y > fmax(q0.second, q1.second))){
420  return false;
421  }
422  }
423 
424  // does it lands on the polygon segment's projection?
425  if((x < fmin(p0.first, p1.first)||(x > fmax(p0.first, p1.first)))
426  &&((y < fmin(p0.second, p1.second)||(y > fmax(p0.second, p1.second))))){
427  return false;
428  }
